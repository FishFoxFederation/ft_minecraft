321 std::vector<char> readFile(const std::string & filename)
322 {
323 std::ifstream file
324 {
325 filename,
326 std::ios::ate | std::ios::binary
327 };
328
329 if (!file.is_open())
330 {
331 throw std::runtime_error("Failed to open file: " + filename);
332 }
333
334 size_t file_size = static_cast<size_t>(file.tellg());
335 std::vector<char> buffer(file_size);
336
337 file.seekg(0);
338 file.read(buffer.data(), file_size);
339 file.close();
340
