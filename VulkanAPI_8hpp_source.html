86 attributeDescriptions[1].offset = offsetof(BlockVertex, normal);

91 attributeDescriptions[2].offset = offsetof(BlockVertex, texCoord);

93 attributeDescriptions[3].binding = 0;
94 attributeDescriptions[3].location = 3;

96 attributeDescriptions[3].offset = offsetof(BlockVertex, texLayer);

98 return attributeDescriptions;
99 }
